DiaDaSemana

import java.util.Scanner;

public class DiaDaSemana {
    public static void main(String[] args) {
        Scanner entrada = new Scanner(System.in);

        System.out.print("Digite um número de 1 a 7: ");
        int numero = entrada.nextInt();

        String dia;

        switch (numero) {
            case 1:
                dia = "Domingo";
                break;
            case 2:
                dia = "Segunda-feira";
                break;
            case 3:
                dia = "Terça-feira";
                break;
            case 4:
                dia = "Quarta-feira";
                break;
            case 5:
                dia = "Quinta-feira";
                break;
            case 6:
                dia = "Sexta-feira";
                break;
            case 7:
                dia = "Sábado";
                break;
            default:
                dia = "Número inválido! Digite de 1 a 7.";
                break;
        }

        System.out.println("O dia da semana é: " + dia);

        entrada.close();
    }
}
